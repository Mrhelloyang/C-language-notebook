两整数相除，取整数，想要小数必须有一个是小数；
#include<stdio.h>
int main()
{
	int b=5;
	int c=2;
	a=b/c;//此处应该为a=b/(c*1.0);因为除和乘的优先级相同，从左到右计算需要加括号；
	printf("a=%lf\n",a);
	return 0;
}


#include<stdio.h>
int main()
{
	int c;
	double a;
	a=(double)c/2;//(double)c没有把c的类型改变只是把c的数取出来，把数做了类型转换；
	printf("a=%lf\n",a);
	return 0;
}



前置和后置的区别:
#include<stdio.h>
int main()
{
	int a,i;
	a=++i;//前置++先加后用
	print("a=%d,i=%d\n",a,i);
	a=i++;//后置++先用后加
	print("a=%d,i=%d\n",a,i);
	return 0;
}
//可以先看++在前在后，在前就先加，在后就后加

逻辑运算符:
&&左右两边都为真才为真；
#include<stdio.h>
int main()
{

	printf("%d\n",!(3==4));//!是非
	printf("%d\n",-1&&-2)；//-1为0为假
	return 0;
}
//||两边为假时才为假
